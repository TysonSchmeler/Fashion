var fonts = [
  "aAnotherTag",
  "ACampusBold",
  "AConceptoNrBoldItalic",
  "AcuminConcept-Black",
  "AcuminConcept-BlackItalic",
  "AcuminConcept-Bold",
  "Arial-Black",
  "Bahnschrift",
  "Calibri",
  "Candara",
  "DSGreece",
  "Moyenage",
  "Saytag-Regular",
  "Schwabacher-Regular",
  "Corbel",
  "Lombardia",
  "MaclenburgGotisch",
  "LowBackDemoRegular",
  "NirmalaUI",
  "SimSun"
];

// –ù–∞—Å—Ç—Ä–æ–π–∫–∏ –º–∞–∫–µ—Ç–∞
var cols = 2;
var rows = 10;
var spacingX = 400;   // —É–≤–µ–ª–∏—á–µ–Ω–æ –¥–ª—è –±–æ–ª–µ–µ –¥–ª–∏–Ω–Ω—ã—Ö –Ω–∞–¥–ø–∏—Å–µ–π
var spacingY = 200;   // —É–≤–µ–ª–∏—á–µ–Ω–æ –¥–ª—è –±–æ–ª–µ–µ –∫—Ä—É–ø–Ω–æ–≥–æ —Ç–µ–∫—Å—Ç–∞
var marginTop = 100;
var marginLeft = 100;

// –†–∞–∑–º–µ—Ä –∞—Ä—Ç–±–æ—Ä–¥–∞
var artboardWidth = spacingX * cols + marginLeft * 2;
var artboardHeight = spacingY * rows + marginTop * 2;
var doc = app.documents.add(DocumentColorSpace.CMYK, artboardWidth, artboardHeight);

// –†–∞–∑–º–µ—Ä —Ç–µ–∫—Å—Ç–∞
var fontSize = 72; // ‚Üê –£–≤–µ–ª–∏—á—å –∑–¥–µ—Å—å —Ä–∞–∑–º–µ—Ä, –µ—Å–ª–∏ —Ö–æ—á–µ—à—å –µ—â—ë –∫—Ä—É–ø–Ω–µ–µ

for (var i = 0; i < fonts.length; i++) {
    var col = i % cols;
    var row = Math.floor(i / cols);

    var posX = marginLeft + col * spacingX;
    var posY = artboardHeight - marginTop - row * spacingY;

    try {
        // –°–æ–∑–¥–∞–µ–º —Ç–µ–∫—Å—Ç
        var textItem = doc.textFrames.add();
        textItem.contents = "SLIZKIY";
        textItem.textRange.characterAttributes.textFont = app.textFonts.getByName(fonts[i]);
        textItem.textRange.characterAttributes.size = fontSize;

        // –ü–æ–∑–∏—Ü–∏–æ–Ω–∏—Ä—É–µ–º
        textItem.position = [posX, posY];

        // –ü—Ä–µ–≤—Ä–∞—â–∞–µ–º –≤ –∫–æ–Ω—Ç—É—Ä—ã
        var outlines = textItem.createOutline();

        // –û–±—ä–µ–¥–∏–Ω—è–µ–º
        outlines.selected = true;
        app.executeMenuCommand("group");
        app.executeMenuCommand("Live Pathfinder Add");
        app.executeMenuCommand("expandStyle");
        app.executeMenuCommand("deselectall");

    } catch (e) {
        alert("–û—à–∏–±–∫–∞ —Å–æ —à—Ä–∏—Ñ—Ç–æ–º: " + fonts[i]);
    }
}

// üîÑ –≠–∫—Å–ø–æ—Ä—Ç –≤ PNG
var exportOptions = new ExportOptionsPNG24();
exportOptions.artBoardClipping = true;
exportOptions.transparency = true;
exportOptions.horizontalScale = 300 / 72 * 100; // 300 dpi
exportOptions.verticalScale = 300 / 72 * 100;

var file = new File(Folder.desktop + "/–ú–∞–∫–µ—Ç—ã_–≤_2_—Å—Ç–æ–ª–±—Ü–∞.png");
doc.exportFile(file, ExportType.PNG24, exportOptions);
